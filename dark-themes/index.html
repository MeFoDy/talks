function setDarkThemeState({ sunset, sunrise }) {
    const now = new Date();
    const toMinutes = (d) => 
        d.getHours() * 60 + d.getMinutes();

    if (toMinutes(now) <= toMinutes(sunrise) || 
        toMinutes(now) >= toMinutes(sunset)) {
        setDarkTheme();
    } else {
        setLightTheme();
    }
}
