Операторы сравнения

  • >, <
  • >=, <=
  • ==, ===
  • !=, !==
12 > 10; // true
({}) != ({}); // true
NaN === NaN; // false

== или ===

  • === не преобразует типы
  • == пытается привести обе стороны к одному типу для сравнения
1 == "1"; // true
1 === "1"; // false

({}) == true; // false
({}).toString() == true; // false
"[object Object]" == "true"; // false
  • Значения null и undefined равны == друг другу и не равны чему бы то ни было ещё.
  • При преобразовании в число null равен 0, а undefined равен NaN.
null == undefined; // true
null === undefined; // false

null > 0; // false
null == 0; // false
null >= 0; // true O_o

Приводите к нужному типу явно!

falsy и truly

  • falsy
    • false
    • 0
    • ''
    • null
    • undefined
    • NaN
  • truly — все остальное

// -------------------------------
var result = true;
if (result) {
    result = 0;
    if (result) {
        result = '';
    }
}
var b = result;
var a = false || 0 || 'value';
a === 'value'; // true
// -------------------------------
var result = false;
if (!result) {
    result = 0;
    if (!result) {
        result = 'value';
    }
}
var a = result;
function test(a, b, c) {
    a = a || 'default';
    b = b || 42;
    c = c || false;
}

Имя функции

  • Если функция что-то делает, вычисляет — начинается с глагола:
    getData, callPolice, joinItechart.
  • Если возвращает какой-то признак — формы глаголов to be или to have:
    isStudent, hasKnowledge.
  • Если функция названа правильно, вам не нужно будет исследовать тело функции в месте её вызова.
